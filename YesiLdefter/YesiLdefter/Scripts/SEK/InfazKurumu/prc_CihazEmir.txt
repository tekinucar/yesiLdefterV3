CREATE PROCEDURE [dbo].[prc_CihazEmir] (@Id int, @FirmId int, @TalepTipiId smallint)  
--ALTER PROCEDURE [dbo].[prc_CihazEmir] (@Id int, @FirmId int, @TalepTipiId smallint)  
AS BEGIN 




if (@Id = 0)
begin
  INSERT INTO [dbo].[CihazEmri]
           ([FirmId]
           ,[IsRun]
           ,[TalepTipiId]
           ,[TarihSaat]
		   ,[RunTarihSaat])
     VALUES
           (@FirmId

begin 
  UPDATE [dbo].[CihazEmri]
   SET [IsRun] = 1
      ,[RunTarihSaat] = getdate()
 WHERE Id = @Id

END
