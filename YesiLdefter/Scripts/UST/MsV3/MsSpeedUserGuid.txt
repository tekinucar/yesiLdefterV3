
begin transaction

 create table MsSpeedUser
 (
    Id              Int IDENTITY(1,1) not null,
    SectorTypeId    SmallInt          not null, 
    PageCode        nVarChar(50)      not null,     
	
    Constraint pk_MsSpeedUser primary key (Id)
 )
 create index X_MsSpeedUser_01 on MsSpeedUser (SectorTypeId)
 create index X_MsSpeedUser_02 on MsSpeedUser (PageCode)
  
 grant select, insert, update, delete on MsSpeedUser to public

commit transaction
