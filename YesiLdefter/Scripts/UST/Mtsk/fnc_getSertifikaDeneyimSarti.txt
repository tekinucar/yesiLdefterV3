CREATE FUNCTION [dbo].[fnc_getSertifikaDeneyimSarti]( @TalepId int)  
RETURNS smallInt
AS   
BEGIN 

  -- [Lkp].[MtskAdayTakipDeneyimSartiTipi] 
  -- 1, 'Deneyim şartı yok'
  -- 2, 'Deneyim şartına uygun'
  -- 3, 'Deneyim şartına uygun değil'
  
  --declare @TalepId int
  --set @TalepId = 7413

  declare @FirmId int = 0
  declare @yMevcutSertifikaTipiId smallInt = 0
  declare @yIstenenSertifikaTipiId smallInt = 0
  declare @isEnAz bit
  declare @deneyimSertifikaTipiId smallInt = 0
  declare @deneyimSartiTipiId smallInt = 0
  declare @deneyimEnAzYilFarki smallInt = 0

  declare @isMevcutOtomatik bit
  declare @isOtomatik bit

  Select 
    @yMevcutSertifikaTipiId = isnull(ins.MevcutSertifikaTipiId,0)
   ,@yIstenenSertifikaTipiId = isnull(ins.IstenenSertifikaTipiId,0)
   ,@isMevcutOtomatik = ins.IsMevcutOtomatik
   ,@isOtomatik = ins.IsOtomatik

  From  MtskAdayTalep as ins, MtskAday as aday
  Where ins.AdayId = aday.Id
  and ins.Id = @TalepId

  Select top 1 
      @isEnAz = isnull(saat.IsEnAz,0)
    , @deneyimSertifikaTipiId = isnull(saat.DeneyimSertifikaTipiId,0) 
    , @deneyimEnAzYilFarki = isnull(saat.DeneyimEnAzYilFarki,0) 
  From MtskSaatUygulama as saat
       left outer join GecerlilikTarihi as gt on ( saat.GecerlilikTarihiId = gt.Id )
  Where saat.IstenenSertifikaTipiId = @yIstenenSertifikaTipiId
  --and   saat.MevcutSertifikaTipiId  = @yMevcutSertifikaTipiId  <<< sakın açma olmaz
  and   gt.KonuTipiId = 21102 -- Uygulama saatleri
  order by gt.Tarih desc
  
  --print '   Mevcut  ehliyet : ' + convert(varchar(5), @yMevcutSertifikaTipiId) 
  --print '   İstenen ehliyet : ' + convert(varchar(5), @yIstenenSertifikaTipiId) 
  --print '   isEnAz : ' + convert(varchar(3), @isEnAz) + ',  @deneyimSartiTipiId : ' + convert(varchar(5), @deneyimSartiTipiId)
  --print '   @isMevcutOtomatik : ' + convert(varchar(3), @isMevcutOtomatik) + ',  @isOtomatik : ' + convert(varchar(5), @isOtomatik)

  -- Deneyim sertifikası yok ise şart yok
  if (@deneyimSertifikaTipiId = 0)
      set @deneyimSartiTipiId = 1 -- Şart yok

  -- Deneyim isteniyor fakat mevcut ehliyet yok ise	   
  if ((@deneyimSertifikaTipiId > 0) and (@yMevcutSertifikaTipiId = 0))
      set @deneyimSartiTipiId = 3 -- uygun değil

  -- print '   isEnAz : ' + convert(varchar(3), @isEnAz) + ',  @deneyimSartiTipiId : ' + convert(varchar(5), @deneyimSartiTipiId)

  if ((@deneyimSertifikaTipiId > 0) and (@isEnAz = 0))
  begin
      if (@deneyimSertifikaTipiId = @yMevcutSertifikaTipiId)
	       set @deneyimSartiTipiId = 2 -- uygun
	  else set @deneyimSartiTipiId = 3 -- uygun değil
  end;

  -- print '   isEnAz : ' + convert(varchar(3), @isEnAz) + ',  @deneyimSartiTipiId : ' + convert(varchar(5), @deneyimSartiTipiId)


  if ((@yMevcutSertifikaTipiId = @yIstenenSertifikaTipiId) and
      (@isMevcutOtomatik = @isOtomatik ))
       set @deneyimSartiTipiId = 3

  -- print '   isEnAz : ' + convert(varchar(3), @isEnAz) + ',  @deneyimSartiTipiId : ' + convert(varchar(5), @deneyimSartiTipiId)

  if ((@deneyimSertifikaTipiId > 0) and (@isEnAz = 1) and (@yMevcutSertifikaTipiId > 0)) 
  begin
      -- kafadan uygun değil dedik
	  set @deneyimSartiTipiId = 3 -- uygun değil
      
	  declare @istenenSertifikaSiraNo smallInt = 0
	  declare @mevcutSertifikaSiraNo  smallInt = 0
	  declare @deneyimSertifikaSiraNo smallInt = 0
	  
      Select @istenenSertifikaSiraNo = isnull(SiraNo,0) from Lkp.MtskSertifikaTipi
	  Where Id = @yIstenenSertifikaTipiId

      Select @mevcutSertifikaSiraNo = isnull(SiraNo,0) from Lkp.MtskSertifikaTipi
	  Where Id = @yMevcutSertifikaTipiId

      Select @deneyimSertifikaSiraNo = isnull(SiraNo,0) from Lkp.MtskSertifikaTipi
	  Where Id = @deneyimSertifikaTipiId

      if ((@mevcutSertifikaSiraNo >= @deneyimSertifikaSiraNo) and
	      (@mevcutSertifikaSiraNo <  @istenenSertifikaSiraNo)) set @deneyimSartiTipiId = 2 -- uygun

       -- print '   @istenenSertifikaSiraNo : ' + convert(varchar(4), @istenenSertifikaSiraNo)
       -- print '   @mevcutSertifikaSiraNo  : ' + convert(varchar(4), @mevcutSertifikaSiraNo)
       -- print '   @deneyimSertifikaSiraNo : ' + convert(varchar(4), @deneyimSertifikaSiraNo)

	  /*
	  declare @deneyimName varchar(5)  
	  Select  @deneyimName = trim(replace(replace(replace(SertifikaTipi, 'Sınıfı Sertifika', ''), '(Manuel)',''), '(Otomatik)',''))
	  from Lkp.MtskSertifikaTipi
	  Where Id = @deneyimSertifikaTipiId

	  --print '   @deneyimName = >' + @deneyimName + '<'
	  
      declare @serId int
      declare @serName varchar(30)
      declare serCursor CURSOR
      for 
      Select Id, SertifikaTipi
      from Lkp.MtskSertifikaTipi
      Where Id > 0

      INTO @serId, @serName;

      begin
	      if (charindex(@deneyimName, @serName) > 0)-- and (charindex(@deneyimName, @serName) < 4))
          begin 
	          --print '   ' + convert(varchar(5), @serId) + '          ' + @serName
              if (@serId = @yMevcutSertifikaTipiId)
			      set @deneyimSartiTipiId = 2 -- uygun 
          end -- if
          --print  '   Uygunluk : ' + convert(varchar(2), @deneyimSartiTipiId) +  '  ' + @serName           
		  if (@deneyimSartiTipiId = 2) break;
		  FETCH NEXT FROM serCursor INTO @serId, @serName;
      end -- While

      Close serCursor;
      DealLocate serCUrsor;
	  */
  end -- if ((@deneyimSertifikaTipiId > 0) and (@isEnAz = 0))

  -- Id  DeneyimSartiTipi
  -- 1   Şart yok
  -- 2   Uygun
  -- 3   Uygun değil

  -- print '   Sonuc : ' + convert(varchar(2), @deneyimSartiTipiId);

  RETURN @deneyimSartiTipiId;


END; -- function
