
/*CreateProcedure*/

CREATE PROCEDURE [dbo].[prc_MtskAdayTakipUygulamaToplamlari] (@FirmId int, @DonemTipiId int)  
AS BEGIN

--declare @FirmId int = 11

   declare @bugunDonemTipiId int 
   set @bugunDonemTipiId = dbo.fnc_getDonemTipiId('BUGUN', null);

-- uSinavDurumTipi < 9
Select 
   tip.KonuId
 , tip.KonuCaption   
 , tip.uSinavDurumTipi as KonuAciklama
 , tip.FormCaption as Islem
 , count(takip.Id) as AdaySayisi 

From dbo.MtskAdayTakip as takip
     left outer join Lkp.MtskAdayTakipuSinavDurumTipi as tip on 
	   ( takip.uSinavDurumTipiId = tip.Id )
Where takip.FirmId = @FirmId
and   takip.KayitTipiId < 4
and   takip.uSinavDurumTipiId < 9  -- 9, 'Başarılı, sertifika aşamasına geçti'
and   takip.Asama1TipiId = 600

Group by 
   tip.KonuId
 , tip.KonuCaption   
 , tip.uSinavDurumTipi
 , tip.FormCaption 

union all

-- uSinavDurumTipi = 9
Select 
   tip.KonuId
 , tip.KonuCaption   
 , tip.uSinavDurumTipi as KonuAciklama
 , tip.FormCaption as Islem
 , count(takip.Id) as AdaySayisi 

From dbo.MtskSinavUygulama as sinav
     left outer join dbo.MtskAdayTakip as takip on ( sinav.TalepId = takip.TalepId )
     left outer join Lkp.MtskAdayTakipuSinavDurumTipi as tip on ( takip.uSinavDurumTipiId = tip.Id )
Where takip.FirmId = @FirmId
and   takip.KayitTipiId <= 4
and   takip.uSinavDurumTipiId = 9  -- 9, 'Başarılı, sertifika aşamasına geçti'
and   takip.Asama1TipiId = 810
and   sinav.DurumuTipiId > 0
and   sinav.SinavDonemTipiId = @bugunDonemTipiId

Group by 
   tip.KonuId
 , tip.KonuCaption   
 , tip.uSinavDurumTipi
 , tip.FormCaption 

union all

-- uSinavDurumTipi = 10 -- Sınav hakkı doldu
Select 
   tip.KonuId
 , tip.KonuCaption   
 , tip.uSinavDurumTipi as KonuAciklama
 , tip.FormCaption as Islem
 , count(takip.Id) as AdaySayisi 

From dbo.MtskSinavUygulama as sinav
     left outer join dbo.MtskAdayTakip as takip on ( sinav.TalepId = takip.TalepId )
     left outer join Lkp.MtskAdayTakipuSinavDurumTipi as tip on ( takip.uSinavDurumTipiId = tip.Id )
Where takip.FirmId = @FirmId
--and   takip.KayitTipiId <= 4
and   takip.uSinavDurumTipiId = 10  -- 10, 'Uygulama sınav hakkı doldu'
and   takip.Asama1TipiId = 820
and   sinav.DurumuTipiId > 0
and   sinav.SinavDonemTipiId = @bugunDonemTipiId

Group by 
   tip.KonuId
 , tip.KonuCaption   
 , tip.uSinavDurumTipi
 , tip.FormCaption 

union all


-- uSinavSonucTipi : Başarılı
Select 
   tip.KonuId
 , tip.KonuCaption   
 , tip.uSinavSonucTipi as KonuAciklama
 , tip.FormCaption as Islem
 , count(sinav.Id) as AdaySayisi 

From dbo.MtskAdayTakip as takip
     left outer join dbo.MtskSinavUygulama as sinav on ( takip.TalepId = sinav.TalepId and takip.uSinavNoGirdigi = sinav.SinavNo )
     left outer join Lkp.MtskAdayTakipuSinavSonucTipi as tip on ( tip.Id = 1 )
Where takip.FirmId = @FirmId
and   takip.KayitTipiId <= 4 
and   takip.Asama1TipiId = 810
and   sinav.PuanTipiId = 100 -- Başarılı
and   sinav.SinavDonemTipiId = @bugunDonemTipiId

Group by 
   tip.KonuId
 , tip.KonuCaption   
 , tip.uSinavSonucTipi
 , tip.FormCaption 

union all

-- uSinavSonucTipi : Başarısız
Select 
   tip.KonuId
 , tip.KonuCaption   
 , tip.uSinavSonucTipi as KonuAciklama
 , tip.FormCaption as Islem
 , count(sinav.Id) as AdaySayisi 

From dbo.MtskAdayTakip as takip
     left outer join dbo.MtskSinavUygulama as sinav on ( takip.TalepId = sinav.TalepId and takip.uSinavNoGirdigi = sinav.SinavNo )
     left outer join Lkp.MtskAdayTakipuSinavSonucTipi as tip on ( tip.Id = 2 )
Where sinav.FirmId = @FirmId
and   sinav.PuanTipiId = 1 -- Başarısız
and   takip.KayitTipiId < 4
and   takip.Asama1TipiId = 600
--and   sinav.SinavDonemTipiId = @bugunDonemTipiId

Group by 
   tip.KonuId
 , tip.KonuCaption   
 , tip.uSinavSonucTipi
 , tip.FormCaption 

union all

-- uSinavSonucTipi : Girmedi
Select 
   tip.KonuId
 , tip.KonuCaption   
 , tip.uSinavSonucTipi as KonuAciklama
 , tip.FormCaption as Islem
 , count(sinav.Id) as AdaySayisi 

From dbo.MtskAdayTakip as takip
     left outer join dbo.MtskSinavUygulama as sinav on ( takip.TalepId = sinav.TalepId and takip.uSinavNoGirdigi = sinav.SinavNo )
     left outer join Lkp.MtskAdayTakipuSinavSonucTipi as tip on ( tip.Id = 3 )
Where sinav.FirmId = @FirmId
and   sinav.PuanTipiId = -1 -- Girmedi
and   takip.KayitTipiId < 4
and   takip.Asama1TipiId = 600
--and   sinav.SinavDonemTipiId = @bugunDonemTipiId

Group by 
   tip.KonuId
 , tip.KonuCaption   
 , tip.uSinavSonucTipi
 , tip.FormCaption 

union all


-- uSinavSonrasiTipi
Select 
   tip.KonuId
 , tip.KonuCaption   
 , tip.uSinavSonrasiTipi as KonuAciklama
 , tip.FormCaption as Islem
 , count(takip.Id) as AdaySayisi 

From dbo.MtskAdayTakip as takip
     left outer join dbo.MtskSinavUygulama as sinav on ( takip.TalepId = sinav.TalepId )
     left outer join Lkp.MtskAdayTakipuSinavSonrasiTipi as tip on ( takip.uSinavSonrasiTipiId = tip.Id )
Where takip.FirmId = @FirmId
and   takip.uSinavDetayId = sinav.Id -- takipdeki asamalar son sınavına ait kararlardır. Aynı ay içinde birden fazla sınava girebiliyor
and   takip.KayitTipiId < 4
and   takip.Asama1TipiId = 600

Group by 
   tip.KonuId
 , tip.KonuCaption   
 , tip.uSinavSonrasiTipi
 , tip.FormCaption
