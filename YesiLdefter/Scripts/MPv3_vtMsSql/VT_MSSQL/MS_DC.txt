

begin transaction

 create table MS_DC
 (
	REF_ID              Int IDENTITY(1,1)  NOT NULL,
    DC_CODE             VarChar(30)    NOT NULL,     
    DC_CAPTION          VarChar(50)    NULL,
    DC_ABOUT            VarChar(200)   NULL,
    DC_TYPE             SmallInt       NULL,  
    TROW_TYPE           SmallInt       NULL, 
    EVENT_TYPE          SmallInt       NULL,
    QUESTION            VarChar(100)   NULL,
    FINAL_MESSAGE       VarChar(100)   NULL,
    WORK_TYPE           SmallInt       NULL,   
    DISPLAY_TYPE        SmallInt       NULL,
    TARGET_TABLE_CODE   VarChar(50)    NULL, 
    TARGET_IP_CODE      VarChar(50)    NULL,
    SOURCE_TABLE_CODE	VarChar(50)    NULL, 
    SOURCE_IP_CODE      VarChar(50)    NULL,
    SOURCE_CHECK_FNAME  VarChar(30)    NULL,
    SOURCE_CHECK_VALUE  VarChar(30)    NULL,
    SOURCE_OPERAND_TYPE VarChar(5)     NULL, 
    TARGET_CHECK_FNAME  VarChar(30)    NULL,
    TARGET_CHECK_VALUE  VarChar(30)    NULL,
    TARGET_OPERAND_TYPE VarChar(5)     NULL, 

	Constraint PK_MS_DC primary key (DC_CODE)

 )

 create index X_MS_DC_1 on MS_DC ( DC_CODE )

 grant select, insert, update, delete on MS_DC to public

commit transaction
